    function processImg(file, maxW, callback) {
        const reader = new FileReader();
        reader.onload = (e) => {
            const img = new Image();
            img.onload = () => {
                const canvas = document.createElement('canvas');
                let w = img.width, h = img.height;
                if (w > maxW) { h *= maxW / w; w = maxW; }
                canvas.width = w; canvas.height = h;
                canvas.getContext('2d').drawImage(img, 0, 0, w, h);
                callback(canvas.toDataURL('image/jpeg', 0.6));
            };
            img.src = e.target.result;
        };
        reader.readAsDataURL(file);
    }

    function getCoords(str) {
        if (!str) return null;
        const reg = /@(-?\d+\.\d+),(-?\d+\.\d+)/;
        const match = str.match(reg);
        if (match) return { lat: parseFloat(match[1]), lon: parseFloat(match[2]) };
        const parts = str.split(',');
        if (parts.length >= 2) {
            const lat = parseFloat(parts[0]);
            const lon = parseFloat(parts[1]);
            if (!isNaN(lat) && !isNaN(lon)) return { lat, lon };
        }
        return null;
    }

    function calcDist(c1, c2) {
        const R = 6371; // km
        const dLat = (c2.lat - c1.lat) * Math.PI / 180;
        const dLon = (c2.lon - c1.lon) * Math.PI / 180;
        const a = Math.sin(dLat/2) * Math.sin(dLat/2) +
                  Math.cos(c1.lat * Math.PI / 180) * Math.cos(c2.lat * Math.PI / 180) *
                  Math.sin(dLon/2) * Math.sin(dLon/2);
        const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
        return R * c;
    }

    function detectarUbicacionProp(i) {
        if (!navigator.geolocation) return alert("Geolocalización no soportada");
        navigator.geolocation.getCurrentPosition(pos => {
            const lat = pos.coords.latitude;
            const lon = pos.coords.longitude;
            voxiData.props[i].maps = `${lat},${lon}`;
            renderProps();
            alert("✅ Ubicación detectada y guardada.");
        }, () => alert("No se pudo detectar la ubicación. Activa el GPS."));
    }

    async function buscarCercanos(i) {
        const p = voxiData.props[i];
        const coords = getCoords(p.maps);
        const listDiv = document.getElementById(`nearbyList${i}`);
        if (!coords) return alert("Primero guarda la ubicación de la propiedad (coordenadas o link de maps).");
        
        listDiv.style.display = 'block';
        listDiv.innerHTML = '<div style="text-align:center; padding:10px; color:#666;"><i class="fas fa-spinner fa-spin"></i> Buscando lugares en Acayuca...</div>';

        // Definición de puntos clave en Acayuca si falla la API o para rapidez
        const acayucaPoints = [
            { n: 'Centro de Acayuca', lat: 19.8972, lon: -98.8872, icon: '🏙️' },
            { n: 'Presidencia Municipal', lat: 19.8970, lon: -98.8868, icon: '🏛️' },
            { n: 'Cancha Deportiva', lat: 19.8965, lon: -98.8890, icon: '⚽' },
            { n: 'Centro de Salud', lat: 19.8980, lon: -98.8850, icon: '🏥' }
        ];

        try {
            // Overpass API Query
            const query = `[out:json];(
                node["amenity"~"school|clinic|hospital|fuel|bank|atm|marketplace|pharmacy"](around:2500,${coords.lat},${coords.lon});
                node["shop"~"convenience|supermarket"](around:2500,${coords.lat},${coords.lon});
            );out body;`;
            const res = await fetch(`https://overpass-api.de/api/interpreter?data=${encodeURIComponent(query)}`);
            const data = await res.json();
            
            let results = data.elements.map(e => ({
                n: e.tags.name || e.tags.amenity || e.tags.shop || 'Lugar cercano',
                lat: e.lat, lon: e.lon,
                icon: e.tags.amenity === 'school' ? '🏫' : 
                      (e.tags.amenity === 'clinic' || e.tags.amenity === 'hospital') ? '🏥' :
                      e.tags.amenity === 'fuel' ? '⛽' :
                      (e.tags.amenity === 'bank' || e.tags.amenity === 'atm') ? '🏦' :
                      (e.tags.shop === 'convenience' || e.tags.shop === 'supermarket') ? '🛒' : '📍'
            }));

            // Agregar puntos fijos de Acayuca
            acayucaPoints.forEach(pt => {
                if (calcDist(coords, pt) < 5) results.push(pt);
            });

            // Ordenar por distancia
            results.forEach(r => {
                r.dist = calcDist(coords, r);
                r.time = Math.round(r.dist * 12); // aprox 5km/h caminando = 12 min/km
            });
            results.sort((a,b) => a.dist - b.dist);

            if (results.length === 0) {
                listDiv.innerHTML = '<div style="padding:10px; text-align:center;">No se encontraron lugares cercanos automáticos. Prueba la consulta manual.</div>';
            } else {
                listDiv.innerHTML = `<div style="font-weight:bold; margin-bottom:10px; font-size:0.8em; color:var(--primary);">LUGARES ENCONTRADOS:</div>`;
                results.slice(0, 10).forEach(r => {
                    const distStr = r.dist < 1 ? `${Math.round(r.dist*1000)}m` : `${r.dist.toFixed(1)}km`;
                    listDiv.innerHTML += `
                        <div class="nearby-item">
                            <div style="display:flex; align-items:center;">
                                <span class="nearby-icon">${r.icon}</span>
                                <span>${r.n}</span>
                            </div>
                            <div>
                                <span class="distance-tag">${distStr}</span>
                                <span class="time-tag">⏱️ ${r.time} min</span>
                            </div>
                        </div>
                    `;
                });
            }
        } catch(e) {
            listDiv.innerHTML = '<div style="padding:10px; color:red;">Error de conexión. Intenta de nuevo.</div>';
        }
    }

    async function consultarDistancia(i) {
        const p = voxiData.props[i];
        const coords = getCoords(p.maps);
        const query = document.getElementById(`queryLoc${i}`).value;
        const resDiv = document.getElementById(`queryRes${i}`);
        if (!coords || !query) return alert("Ingresa ubicación y búsqueda.");

        resDiv.innerHTML = "🔍 Buscando...";
        try {
            const overQuery = `[out:json];node["name"~"${query}",i](around:5000,${coords.lat},${coords.lon});out body;`;
            const res = await fetch(`https://overpass-api.de/api/interpreter?data=${encodeURIComponent(overQuery)}`);
            const data = await res.json();
            
            if (data.elements.length > 0) {
                const e = data.elements[0];
                const d = calcDist(coords, {lat: e.lat, lon: e.lon});
                const t = Math.round(d * 12);
                resDiv.innerHTML = `<span style="color:var(--success);">✅ ${e.tags.name || query} está a ${d.toFixed(2)} km (${t} min aprox).</span><br>
                <a href="https://www.google.com/maps/dir/${coords.lat},${coords.lon}/${e.lat},${e.lon}" target="_blank" style="color:var(--info); font-size:0.8em;">🗺️ Ver ruta en Maps</a>`;
            } else {
                resDiv.innerHTML = `<span style="color:var(--primary);">❌ No encontré "${query}" cerca de aquí.</span>`;
            }
        } catch(e) { resDiv.innerHTML = "Error al consultar."; }
    }

    async function descargarCroquis() {
        const area = document.getElementById('captureArea');
        const canvas = await html2canvas(area);
        const link = document.createElement('a');
        link.download = 'croquis_voxi.jpg';
        link.href = canvas.toDataURL('image/jpeg', 0.9);

    async function compartirCroquisWA(titulo) {
        alert("Generando imagen... espera un momento.");
        const area = document.getElementById('captureArea');
        const canvas = await html2canvas(area);
        canvas.toBlob(blob => {
            const file = new File([blob], "croquis.jpg", { type: "image/jpeg" });
            if (navigator.canShare && navigator.canShare({ files: [file] })) {
                navigator.share({
                    files: [file],
                    title: 'Croquis: ' + titulo,
                    text: 'Te comparto la ubicación de la propiedad en Acayuca.'
                });
            } else {
                alert("Tu navegador no soporta compartir archivos directamente. Usa el botón GUARDAR e inserta la foto en WhatsApp.");
            }
        }, 'image/jpeg');
    }
